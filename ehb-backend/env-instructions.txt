# Backend Environment Setup (.env)
# Create a file named .env in this directory with these contents:

NODE_ENV=development
PORT=5000

# Frontend Environment Setup (.env.local)
# Create a file named .env.local in the root directory with these contents:

NEXT_PUBLIC_API_URL=http://localhost:5000/api
NEXTAUTH_URL=http://localhost:3000
NEXTAUTH_SECRET=ehb-nextauth-secret-key-2024
